78  template <typename length>
79  typename std::enable_if<std::is_arithmetic<length>::value, length>::type
80  pitagoras_diagonais(length largest_diagonal, length smallest_diagonal)
81  {
82  if (largest_diagonal < smallest_diagonal)
83  {
84  throw std::invalid_argument("largest_diagonal parameter cannot be smaller than smallest_diagonal parameter");
85  }
86 
87  return sqrt((largest_diagonal / 2) * (largest_diagonal / 2) +
88  (smallest_diagonal / 2) * (smallest_diagonal / 2));
89  }
